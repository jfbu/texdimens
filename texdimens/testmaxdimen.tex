\documentclass{article}
\usepackage{texdimens}
\usepackage{xinttools}
\begin{document}
\immediate\write128{IGNORE "maxdimen is realized..." in output after
 "! Dimension too large" error message}
\immediate\write128{OK? (hit <return>)}
\read-1to\x
\nonstopmode
\xintFor #1 in {
    bp,
    nd,
    dd,
    mm,
    pc,
    nc,
    cc,
    cm,
    in}
:
{%
\edef\x{\csname texdimin#1\endcsname\maxdimen}%
\immediate\write128{#1 -> \x}%
% is \maxdimen attainable ? 
% This may raise Dimension too large, ignore terminal output then
\ifdim\x #1=\maxdimen\immediate\write128{maxdimen is realized as \x#1}\else
% here we know that \maxdimen is not attainable and that \x #1 gives maximal
% attainable dimension
\immediate\write128{maxdimen (\number\maxdimen) is not attainable with unit "#1"}%
\immediate\write128{maximal attainable one is \x#1 (\number\dimexpr\x#1)}%
\fi
}
% This showed that \maxdimen is attainable only for bp and nd
% We still need to recover maximal attainable for dd, nc, and in
\xintFor #1 in {
%    bp,
%    nd,
    dd,
%    mm,
%    pc,
    nc,
%    cc,
%    cm,
    in}
:
{%
\edef\x{\csname texdimin#1\endcsname\maxdimen}%
\immediate\write128{\x#1 > maxdimen}%
\edef\y{\number\dimexpr\x pt}% recover the computed R = N + 1 (as Ruu
                             % triggered Dimension too large)
\immediate\write128{y est \y}%
\edef\z{\the\numexpr\y-1}%
\edef\x{\texdiminpt{\z sp}}%
\immediate\write128{maxdimen (\number\maxdimen) is not attainable with unit "#1"}%
\immediate\write128{maximal attainable one is \x#1 (\number\dimexpr\x#1)}%
}

\end{document}
