\input texdimens

\newwrite\out
\immediate\openout\out=\jobname-out.txt

\immediate\write\out{}

\def\test#1{%

\immediate\write\out{#1 (\the\numexpr\dimexpr#1\relax\relax)}

\edef\x{\texdimenbothincm{#1}}
\immediate\write\out{\x cm, \the\numexpr \dimexpr \x cm<=\the\numexpr\dimexpr#1, \the\dimexpr \x cm\relax}

\edef\y{\texdimenbothcmin{#1}}
\immediate\write\out{\y in, \the\numexpr \dimexpr \y in<=\the\numexpr\dimexpr#1, \the\dimexpr \y in\relax}

\edef\z{\texdimenbothincmpt{#1}}
\immediate\write\out{\z pt, \the\numexpr \dimexpr \z pt<=\the\numexpr\dimexpr#1, \the\dimexpr \z pt\relax}

\edef\x{\texdimenbothincm{-#1}}
\immediate\write\out{\x cm, \the\numexpr \dimexpr \x cm>=\the\numexpr\dimexpr-#1, \the\dimexpr \x cm\relax}

\edef\y{\texdimenbothcmin{-#1}}
\immediate\write\out{\y in, \the\numexpr \dimexpr \y in>=\the\numexpr\dimexpr-#1, \the\dimexpr \y in\relax}

\edef\z{\texdimenbothincmpt{-#1}}
\immediate\write\out{\z pt, \the\numexpr \dimexpr \z pt>=\the\numexpr\dimexpr-#1, \the\dimexpr \z pt\relax}

\immediate\write\out{}
}

\test\maxdimen

\test{1000sp}

\test{3612sp}

\test{3613sp}

\test{5000sp}

\test{7226sp}

\test{7227sp}

\test{10000sp}

\test{11000sp}

\test{15000sp}

\test{100000sp}

\test{1000000sp}

\test{1cm}

\test{1in}

\test{127cm}

\test{50in}

\bye
